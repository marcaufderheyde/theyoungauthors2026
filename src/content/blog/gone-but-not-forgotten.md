<!-- wp:paragraph -->
<p>When the wind shakes the shutters outside&nbsp;</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>I crawl to the couch with my cup of coffee.</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>Staring through the panes at the pale sick sky,</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>I nustle the warmth off the fabric around me.</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>No such comfort exists outside of dry socks on cold feet.</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>The thought of work hangs heavy in my mind</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>Yet still I make space in my time</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>To search through the clouds</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>Thinking of her.</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>Perhaps I’ll go for a walk in the woods</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>Or maybe I’ll play the same notes through my head.</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>Whatever I do,</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>Wherever I go,</p>
<!-- /wp:paragraph -->

<!-- wp:paragraph -->
<p>I’ll never stop plucking her hair from my pillow.</p>
<!-- /wp:paragraph -->
